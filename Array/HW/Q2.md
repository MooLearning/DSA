## 🎀 C++ Code 🎀



```C++

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

class MinMax {

public:

    static void findMinMax(vector<int> arr) {

        if (arr.empty()) {
            cout << "Invalid Array\n";
            return;
        }

        int minValue = arr[0];
        int maxValue = arr[0];

        for (int value : arr) {

            if (value < minValue)
                minValue = value;

            if (value > maxValue)
                maxValue = value;
        }

        cout << "Minimum: " << minValue << endl;
        cout << "Maximum: " << maxValue << endl;
    }


    static void optimizedMinMax(vector<int> arr) {

        if (arr.empty()) {
            cout << "Invalid Array\n";
            return;
        }

        cout << "Minimum: " << *min_element(arr.begin(), arr.end()) << endl;
        cout << "Maximum: " << *max_element(arr.begin(), arr.end()) << endl;
    }
};

int main() {

    vector<vector<int>> testCases = {
        {12,5,27,3,19},
        {10},
        {},
        {-1,-5,-3},
        {5,5,5}
    };

    for (auto arr : testCases) {

        cout << endl;

        MinMax::findMinMax(arr);

        cout << "Optimized:\n";

        MinMax::optimizedMinMax(arr);
    }

    return 0;
}
```

```java

import java.util.Arrays;

class MinMax {

    // Manual
    static void findMinMax(int[] array) {

        if (array == null || array.length == 0) {
            System.out.println("Invalid Array");
            return;
        }

        int min = array[0];
        int max = array[0];

        for (int value : array) {

            if (value < min) {
                min = value;
            }

            if (value > max) {
                max = value;
            }
        }

        System.out.println("Minimum: " + min);
        System.out.println("Maximum: " + max);
    }


    // Optimized
    static void optimizedMinMax(int[] array) {

        if (array == null || array.length == 0) {
            System.out.println("Invalid Array");
            return;
        }

        System.out.println("Minimum: " + Arrays.stream(array).min().getAsInt());
        System.out.println("Maximum: " + Arrays.stream(array).max().getAsInt());
    }


    public static void main(String[] args) {

        int[][] testCases = {
            {12,5,27,3,19},
            {10},
            {},
            {-1,-5,-3},
            {5,5,5}
        };

        for (int[] arr : testCases) {

            System.out.println();

            findMinMax(arr);

            System.out.println("Optimized:");

            optimizedMinMax(arr);
        }
    }
}
```
