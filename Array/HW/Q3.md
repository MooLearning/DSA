# 📃 Count even and odd values
---
## ❓ Question / Description

Given an array, count how many values are even and how many are odd.



| Input | Output |
|:-------------|:--------------|
| `[10, 15, 20, 33, 42]` | Even: 3 <br> Odd: 2 |

## 🎀 C++ Code 🎀



```C++

#include <iostream>
#include <vector>
using namespace std;

class EvenOddCounter {

public:

    static void countEvenOdd(vector<int> arr) {

        if (arr.empty()) {
            cout << "Invalid Array\n";
            return;
        }

        int even = 0;
        int odd = 0;

        for (int value : arr) {

            if (value % 2 == 0)
                even++;
            else
                odd++;
        }

        cout << "Even: " << even << endl;
        cout << "Odd: " << odd << endl;
    }


    static void optimizedCount(vector<int> arr) {

        if (arr.empty()) {
            cout << "Invalid Array\n";
            return;
        }

        int even = 0;

        for (int value : arr)
            even += (value % 2 == 0);

        int odd = arr.size() - even;

        cout << "Even: " << even << endl;
        cout << "Odd: " << odd << endl;
    }
};

int main() {

    vector<vector<int>> testCases = {
        {10,15,20,33,42},
        {2,4,6},
        {1,3,5},
        {},
        {-2,-3,-4}
    };

    for (auto arr : testCases) {

        cout << endl;

        EvenOddCounter::countEvenOdd(arr);

        cout << "Optimized:\n";

        EvenOddCounter::optimizedCount(arr);
    }

    return 0;
}
```

```java
class EvenOddCounter {

    // Manual
    static void countEvenOdd(int[] array) {

        if (array == null || array.length == 0) {
            System.out.println("Invalid Array");
            return;
        }

        int even = 0;
        int odd = 0;

        for (int value : array) {

            if (value % 2 == 0) {
                even++;
            } else {
                odd++;
            }
        }

        System.out.println("Even: " + even);
        System.out.println("Odd: " + odd);
    }


    // Optimized
    static void optimizedCount(int[] array) {

        if (array == null || array.length == 0) {
            System.out.println("Invalid Array");
            return;
        }

        int even = 0;

        for (int value : array) {
            even += (value % 2 == 0) ? 1 : 0;
        }

        int odd = array.length - even;

        System.out.println("Even: " + even);
        System.out.println("Odd: " + odd);
    }


    public static void main(String[] args) {

        int[][] testCases = {
            {10,15,20,33,42},
            {2,4,6},
            {1,3,5},
            {},
            {-2,-3,-4}
        };

        for (int[] arr : testCases) {

            System.out.println();

            countEvenOdd(arr);

            System.out.println("Optimized:");

            optimizedCount(arr);
        }
    }
}
```

```Python
class EvenOddCounter:

    # Manual
    @staticmethod
    def count_even_odd(array):

        if array is None or len(array) == 0:
            print("Invalid Array")
            return

        even = 0
        odd = 0

        for value in array:

            if value % 2 == 0:
                even += 1
            else:
                odd += 1

        print("Even:", even)
        print("Odd:", odd)


    # Optimized
    @staticmethod
    def optimized_count(array):

        if array is None or len(array) == 0:
            print("Invalid Array")
            return

        even = sum(value % 2 == 0 for value in array)
        odd = len(array) - even

        print("Even:", even)
        print("Odd:", odd)


test_cases = [
    [10,15,20,33,42],
    [2,4,6],
    [1,3,5],
    [],
    [-2,-3,-4]
]

for arr in test_cases:

    print()

    EvenOddCounter.count_even_odd(arr)

    print("Optimized:")

    EvenOddCounter.optimized_count(arr)
```
