# 📃 Print elements

---

## ❓ Question / Description

Given an array, print all elements from left to right and then from right to left.

| Input | Output |
|:-------------|:--------------|
| `[10, 20, 30, 40, 50]` | Forward: 10 20 30 40 50 <br> Backward: 50 40 30 20 10 |

```C++

#include <iostream>
#include <vector>
using namespace std;

class ArrayPrint {

public:

    // =========================
    // Manual Methods
    // =========================

    static void forwardPrint(vector<int> arr) {

        if (arr.empty()) {
            cout << "Empty Array\n";
            return;
        }

        for (int value : arr) {
            cout << value << " ";
        }

        cout << endl;
    }

    static void backwardPrint(vector<int> arr) {

        if (arr.empty()) {
            cout << "Empty Array\n";
            return;
        }

        for (int i = arr.size() - 1; i >= 0; i--) {
            cout << arr[i] << " ";
        }

        cout << endl;
    }


    // =========================
    // Optimized Method
    // =========================

    static void printArray(vector<int> arr, bool reverse = false) {

        if (arr.empty()) {
            cout << "Empty Array\n";
            return;
        }

        if (!reverse) {

            for (int value : arr) {
                cout << value << " ";
            }

        } else {

            for (int i = arr.size() - 1; i >= 0; i--) {
                cout << arr[i] << " ";
            }
        }

        cout << endl;
    }
};

int main() {

    vector<vector<int>> testCases = {
        {1,2,3,4,5},
        {10},
        {},
        {-1,-2,-3},
        {5,5,5,5}
    };

    for (int i = 0; i < testCases.size(); i++) {

        cout << "\nTest Case " << i + 1 << endl;

        ArrayPrint::forwardPrint(testCases[i]);
        ArrayPrint::backwardPrint(testCases[i]);

        cout << "Optimized:\n";

        ArrayPrint::printArray(testCases[i]);
        ArrayPrint::printArray(testCases[i], true);
    }

    return 0;
}
```

```Python
class ArrayPrint:

    # =========================
    # Manual Methods
    # =========================

    @staticmethod
    def forward_print(array):

        if array is None:
            print("Null Array")
            return

        if not array:
            print("Empty Array")
            return

        for value in array:
            print(value, end=" ")

        print()


    @staticmethod
    def backward_print(array):

        if array is None:
            print("Null Array")
            return

        if not array:
            print("Empty Array")
            return

        for i in range(len(array) - 1, -1, -1):
            print(array[i], end=" ")

        print()


    # =========================
    # Optimized Method
    # =========================

    @staticmethod
    def print_array(array, reverse=False):

        if array is None:
            print("Null Array")
            return

        if not array:
            print("Empty Array")
            return

        iterable = reversed(array) if reverse else array

        for value in iterable:
            print(value, end=" ")

        print()


# =========================
# Test Cases
# =========================

test_cases = [

    [5,5,5,5]
]

for index, arr in enumerate(test_cases, start=1):

    ArrayPrint.forward_print(arr)
    ArrayPrint.backward_print(arr)

    print("Optimized:")

    ArrayPrint.print_array(arr)
    ArrayPrint.print_array(arr, True)

print("\nNull Test Case")

ArrayPrint.forward_print(None)
```
